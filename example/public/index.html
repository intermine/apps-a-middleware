Dendrogram Clustering


// Use InterMine API Loader to fetch FatApps client.
intermine.load('fatapps', function(err) {
    // Instantiate the library pointing to a service.
    var client = new intermine.fatApps('http://intermine-fatapps.labs.intermine.org');
    // Load a specific app after satisfying its dependencies; passing target & extra config.
    client.load('mouse-phenotype-dendrogram', '#dendrogram', { 'width': 790, 'height': 600, 'symbol': 'MGI:97747' });
});
