Dendrogram Clustering


// give us report widgets from a specific service in a callback `widgets`
intermine.load('reportWidgets', 'http://reportwidgets-intermine.rhcloud.com', function(widgets) {
    // now load a specific widget and place it in a target element passing it extra config.
    widgets.load('mouse-phenotype-dendrogram', '#dendrogram', { 'width': 790, 'height': 600, 'symbol': 'MGI:97747' });
});
